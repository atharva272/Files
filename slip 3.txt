import java.util.Observable;
import java.util.Observer;

// WeatherData class (Subject)
class WeatherData extends Observable {
    private float temperature;
    private float humidity;
    private float pressure;

    public float getTemperature() {
        return temperature;
    }
    public float getHumidity() {
        return humidity;
    }
    public float getPressure() {
        return pressure;
    }

    public void setMeasurements(float temperature, float humidity, float pressure) {
        this.temperature = temperature;
        this.humidity = humidity;
        this.pressure = pressure;
        measurementsChanged();
    }

    public void measurementsChanged() {
        setChanged();        // mark as updated
        notifyObservers();   // notify all observers
    }
}

// Display class (Observer)
class CurrentDisplay implements Observer {

    @Override
    public void update(Observable o, Object arg) {
        WeatherData wd = (WeatherData) o;
        System.out.println("Updated Weather: Temp=" 
            + wd.getTemperature() + "Â°C, Humidity=" 
            + wd.getHumidity() + "%, Pressure=" 
            + wd.getPressure() + " hPa");
    }
}

// Main Class
public class Main {
    public static void main(String[] args) {

        WeatherData weatherData = new WeatherData();
        CurrentDisplay display = new CurrentDisplay();

        weatherData.addObserver(display);

        // First update
        weatherData.setMeasurements(30.5f, 75.2f, 1012.3f);

        // Second update
        weatherData.setMeasurements(32.0f, 80.0f, 1010.0f);
    }
}

#include <LiquidCrystal.h>
LiquidCrystal lcd(13, 12, 11, 10, 9, 8); 
const int SensorPin = 2;
const int LEDPin = 7;
void setup() {
  pinMode(SensorPin, INPUT); 
  pinMode(LEDPin, OUTPUT); 
  lcd.begin(20, 4); 
  lcd.print("Hello");
  lcd.setCursor(0,0); 
  lcd.print("  THE BRIGHT LIGHT    ");
  lcd.setCursor(0,1);
  lcd.print("  VIBRATION SENSOR");
}
void loop() {
  if(digitalRead(SensorPin) == HIGH){
    digitalWrite(LEDPin,HIGH);
    lcd.setCursor(0,3); 
    lcd.print(" Vibration Detected    "); 
    delay(50);
    lcd.setCursor(0,3); 
    lcd.print(" Vibration Detected.    "); 
    delay(70);
    lcd.setCursor(0,3); 
    lcd.print(" Vibration Detected..   "); 
    delay(70);
    lcd.setCursor(0,3); 
    lcd.print(" Vibration Detected...   "); 
    delay(70); 
}
  else{
    digitalWrite(LEDPin,LOW);
    lcd.setCursor(0,3);  lcd.print("    No Vibration       "); 
   }
}
