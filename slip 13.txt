// ----- Volt Class -----
class Volt {
    private int volts;

    public Volt(int volts) {
        this.volts = volts;
    }

    public int getVolts() {
        return volts;
    }

    public void setVolts(int volts) {
        this.volts = volts;
    }
}

// ----- Socket (Adaptee) producing 120V -----
class Socket {
    public Volt getVolt() {
        return new Volt(120);
    }
}

// ----- Adapter Interface (Target) -----
interface SocketAdapter {
    Volt get3Volt();
    Volt get12Volt();
    Volt get120Volt();
}

// ----- Class Adapter (extends Socket + implements adapter) -----
class SocketAdapterImpl extends Socket implements SocketAdapter {

    @Override
    public Volt get3Volt() {
        return convertVolt(getVolt(), 40); // 120/40 = 3V
    }

    @Override
    public Volt get12Volt() {
        return convertVolt(getVolt(), 10); // 120/10 = 12V
    }

    @Override
    public Volt get120Volt() {
        return getVolt(); // original 120V
    }

    private Volt convertVolt(Volt v, int divisor) {
        return new Volt(v.getVolts() / divisor);
    }
}

// ----- MAIN (Client) -----
public class Main {
    public static void main(String[] args) {

        SocketAdapter adapter = new SocketAdapterImpl();

        System.out.println("Mobile needs: 3V → " + adapter.get3Volt().getVolts() + "V");
        System.out.println("Tablet needs: 12V → " + adapter.get12Volt().getVolts() + "V");
        System.out.println("TV needs: 120V → " + adapter.get120Volt().getVolts() + "V");
    }
}
