import java.util.*;

// Adapter: Converts Enumeration → Iterator
class EnumerationIterator implements Iterator<Object> {

    Enumeration<?> enumeration;

    public EnumerationIterator(Enumeration<?> enumeration) {
        this.enumeration = enumeration;
    }

    @Override
    public boolean hasNext() {
        return enumeration.hasMoreElements();
    }

    @Override
    public Object next() {
        return enumeration.nextElement();
    }

    @Override
    public void remove() {
        throw new UnsupportedOperationException("Remove not supported");
    }
}

public class Main {
    public static void main(String[] args) {

        // Old legacy API → Enumeration
        Vector<String> vector = new Vector<>();
        vector.add("Apple");
        vector.add("Banana");
        vector.add("Mango");

        Enumeration<String> enumeration = vector.elements();

        // Using our Adapter to convert Enumeration → Iterator
        Iterator<Object> iterator = new EnumerationIterator(enumeration);

        System.out.println("Iterating using Iterator but source is Enumeration:");
        while (iterator.hasNext()) {
            System.out.println(iterator.next());
        }
    }
}
