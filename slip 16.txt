import java.util.*;

// ----- Subject -----
class NumberSubject {
    private int number;
    private List<Observer> observers = new ArrayList<>();

    public void attach(Observer o) {
        observers.add(o);
    }

    public void setNumber(int number) {
        this.number = number;
        notifyObservers();
    }

    public int getNumber() {
        return number;
    }

    private void notifyObservers() {
        for (Observer o : observers) {
            o.update();
        }
    }
}

// ----- Observer Interface -----
interface Observer {
    void update();
}

// ----- Concrete Observers -----
class BinaryObserver implements Observer {
    NumberSubject subject;

    public BinaryObserver(NumberSubject subject) {
        this.subject = subject;
        subject.attach(this);
    }

    public void update() {
        System.out.println("Binary: " + Integer.toBinaryString(subject.getNumber()));
    }
}

class OctalObserver implements Observer {
    NumberSubject subject;

    public OctalObserver(NumberSubject subject) {
        this.subject = subject;
        subject.attach(this);
    }

    public void update() {
        System.out.println("Octal: " + Integer.toOctalString(subject.getNumber()));
    }
}

class HexObserver implements Observer {
    NumberSubject subject;

    public HexObserver(NumberSubject subject) {
        this.subject = subject;
        subject.attach(this);
    }

    public void update() {
        System.out.println("Hexadecimal: " + Integer.toHexString(subject.getNumber()).toUpperCase());
    }
}

// ----- MAIN -----
public class Main {
    public static void main(String[] args) {

        NumberSubject subject = new NumberSubject();

        // Add observers
        new BinaryObserver(subject);
        new OctalObserver(subject);
        new HexObserver(subject);

        System.out.println("---- Entering number: 10 ----");
        subject.setNumber(10);

        System.out.println("\n---- Entering number: 100 ----");
        subject.setNumber(100);
    }
}
